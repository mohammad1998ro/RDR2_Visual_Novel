// Exception Handling
// This code safely loads data from a file.
// If an error occurs, an informative message is displayed to the user
// instead of terminating the application.

try
{
    string[] liniiSalvate =
        File.ReadAllLines("salvare.txt");

    pasPoveste =
        int.Parse(liniiSalvate[0]);

    apreciereGang =
        int.Parse(liniiSalvate[1]);
}
catch (Exception ex)
{
    MessageBox.Show(
        "Failed to load save file: " + ex.Message,
        "Error",
        MessageBoxButtons.OK,
        MessageBoxIcon.Error);
}
